# Function to fetch OHLCV data
def fetch_ohlcv(exchange, symbol, since=None):
    if since:
        since_timestamp = int(pd.Timestamp(since).timestamp() * 1000)  # Convert date to milliseconds
        return exchange.fetch_ohlcv(symbol, timeframe='1d', since=since_timestamp)
    else:
        return exchange.fetch_ohlcv(symbol, timeframe='1d')

# Fetch data for each token
def fetch_token_data():
    stader_data = fetch_ohlcv(mexc, 'SD/USDT', '2024-06-01T00:00:00Z')
    ton_data = fetch_ohlcv(kucoin, 'TON/USDT', '2023-12-01T00:00:00Z')
    npc_data = fetch_ohlcv(mexc, 'NPC/USDT', '2024-06-01T00:00:00Z')
    why_data = fetch_ohlcv(xt, 'WHY/USDT', '2024-06-01T00:00:00Z')
    btc_data = fetch_ohlcv(kucoin, 'BTC/USDT', '2023-12-01T00:00:00Z')  # Fetch BTC/USDT data
    return stader_data, ton_data, npc_data, why_data, btc_data

# Convert data to DataFrame for plotting
def data_to_df(data):
    df = pd.DataFrame(data, columns=['timestamp', 'open', 'high', 'low', 'close', 'volume'])
    df['timestamp'] = pd.to_datetime(df['timestamp'], unit='ms')
    df.set_index('timestamp', inplace=True)
    return df

# Calculate percentage change based on purchase price
def calculate_pnl(df, purchase_price):
    return (df['close'] / purchase_price - 1) * 100

# Special case for Stader PNL calculation
def calculate_stader_pnl(purchase_price, exit_price):
    return ((exit_price / purchase_price) - 1) * 100

# Function to plot data
def plot_data(show_symbols=False):
    stader_data, ton_data, npc_data, why_data, btc_data = fetch_token_data()
    
    df_stader = data_to_df(stader_data)
    df_ton = data_to_df(ton_data)
    df_npc = data_to_df(npc_data)
    df_why = data_to_df(why_data)
    df_btc = data_to_df(btc_data)
    
    # Calculate PNL percentages
    stader_pnl = calculate_stader_pnl(0.5955, 0.7986)  # Stader PNL based on buy and exit price
    df_ton['TON_pnl'] = calculate_pnl(df_ton, 2.14)
    df_npc['NPC_pnl'] = calculate_pnl(df_npc, 0.0174)
    df_why['WHY_pnl'] = calculate_pnl(df_why, 0.00000012)
    
    # Calculate BTC/USDT percentage change from December 2023
    df_btc['BTC_pnl'] = calculate_pnl(df_btc, df_btc['close'].iloc[0])
    
    # Combine all PNLs into a single DataFrame
    df_combined = pd.concat([pd.Series(stader_pnl, index=df_stader.index, name='SD_pnl'),
                             df_ton['TON_pnl'],
                             df_npc['NPC_pnl'],

    df_combined.columns = ['SD_pnl', 'TON_pnl', 'NPC_pnl', 'WHY_pnl', 'BTC_pnl']
    
    # Include Stader PNL in Portfolio Calculation
    df_combined['Portfolio_pnl'] = df_combined[['TON_pnl', 'NPC_pnl', 'WHY_pnl']].sum(axis=1) + stader_pnl
    
    plt.figure(figsize=(20, 24))  # Increased height to accommodate more subplots

# Function to create a GUI with an input box instead of a popup
def create_password_prompt():
    # Create the main application window
    root = tk.Tk()
    root.title("Enter Password")
    
    # Set window size and position
    window_width = 300
    window_height = 150
    screen_width = root.winfo_screenwidth()
    screen_height = root.winfo_screenheight()
    position_top = int(screen_height / 2 - window_height / 2)
    position_right = int(screen_width / 2 - window_width / 2)
    root.geometry(f"{window_width}x{window_height}+{position_right}+{position_top}")
    
    # Create a label for the password prompt
    label = ttk.Label(root, text="Enter the password:")
    label.pack(pady=20)
    
    # Create an entry widget for the password
    password_entry = ttk.Entry(root, show='*', width=25)
    password_entry.pack(pady=5)
    
    # Function to handle the password entry and call the plot function
    def on_submit():
        password = password_entry.get()
        correct_password = "your_password_here"  # Set your password here
        
        if password == correct_password:
            root.destroy()  # Close the input window
            plot_data(show_symbols=True)
        else:
            root.destroy()  # Close the input window
            plot_data(show_symbols=False)
    
    # Create a button to submit the password
    submit_button = ttk.Button(root, text="Submit", command=on_submit)
    submit_button.pack(pady=20)
    
    # Start the GUI event loop
    root.mainloop()

if __name__ == "__main__":
    create_password_prompt()
